%%MinTF1
\begin{exercise}
	If $f'(a) = 0$, then $x=a$ is a local extremum of $f$.  
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
        \begin{feedback} Here are two ways this can be false: 1) It isn't $x=a$ that would be the max, but rather $f(a)$.  2) $f(x)$ could be increasing on either side of $x=a$ and so there would not be a maximum at $x=a$
    \end{feedback}
\end{exercise}


%%MinTF4
\begin{exercise}
If $f$ is continuous, $f'(x)<0$ when $x<a$ and $f'(x) \geq 0$ when $x>a$, then $f(a)$ is a local max.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF3
\begin{exercise}
A function defined on a \textbf{closed} interval must have a local extremum.
	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF5
\begin{exercise}
If $f$ has a local max at $x=a$, then its square $f^2$ must also have a local max at $x=a$
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}

\end{exercise}

	\end{multipleChoice}
\end{exercise}

\begin{exercise}

	\begin{multipleChoice}

		\choice{False}
	\end{multipleChoice}
\end{exercise}
